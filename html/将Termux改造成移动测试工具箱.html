    简介

    根文件系统$PREFIX：/data/data/com.termux/files/usr
    家目录$HOME：/data/data/com.termux/files/home

    切换国内源

    Termux镜像源

    sed -i 's@^\(deb.*stable main\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/termux-packages-24 stable main@' $PREFIX/etc/apt/sources.list
    sed -i 's@^\(deb.*games stable\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/game-packages-24 games stable@' $PREFIX/etc/apt/sources.list.d/game.list
    sed -i 's@^\(deb.*science stable\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/science-packages-24 science stable@' $PREFIX/etc/apt/sources.list.d/science.list

    社区源

    root-repo

    sed -i 's@^\(deb.*root stable\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/termux-root-packages-24 root stable@' $PREFIX/etc/apt/root.list

    x11-repo

    sed -i 's@^\(deb.*x11 main\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/x11-packages x11 main@' $PREFIX/etc/apt/sources.list.d/x11.list

    unstable-repo

    sed -i 's@^\(deb.*unstable main\)$@#\1\ndeb https://mirror.tuna.tsinghua.edu.cn/termux/unstable-packages unstable main@' $PREFIX/etc/apt/sources.list.d/unstable.list

    安装工具

    nmap

    pkg install nmap

    Sqlmap

    // 安装python3

    pkg install python

    // 克隆sqlmap仓库

    git clone --depth 1 https://github.com/sqlmapproject/sqlmap.git sqlmap-dev

    Routesploit

    // 克隆soutesploit仓库

    git clone https://www.github.com/threat9/routersploit.git

    // 安装依赖

    python -m pip install -r requirements.txt

    // 启动Routesploit

    python rsf.py

    Metasploit-Framework

    // 添加unstable-repo源并更新缓存

    pkg install unstable-repe && pkg update

    // 安装Metasploit

    pkg install metasploit

    启动Postgresql数据库

    pg_ctl -D /data/data/com.termux/files/usr/var/lib/postgresql -l logfile start

    运行Metasploit

    msfconsole
